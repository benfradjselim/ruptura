import logging
import numpy as np
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error
import matplotlib.pyplot as plt
import seaborn as sns

# Set up logging
logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')

def load_data(file_path: str) -> pd.DataFrame:
    """
    Load data from a CSV file.

    Args:
    file_path (str): Path to the CSV file.

    Returns:
    pd.DataFrame: Loaded data.

    Raises:
    FileNotFoundError: If the file does not exist.
    ValueError: If the file is empty or cannot be parsed.
    """
    try:
        data = pd.read_csv(file_path)
        logging.info(f"Data loaded from {file_path}")
        return data
    except FileNotFoundError as e:
        logging.error(f"File not found: {file_path}")
        raise e
    except pd.errors.EmptyDataError as e:
        logging.error(f"File is empty: {file_path}")
        raise ValueError("File is empty") from e
    except pd.errors.ParserError as e:
        logging.error(f"Error parsing file: {file_path}")
        raise ValueError("Error parsing file") from e

def preprocess_data(data: pd.DataFrame) -> pd.DataFrame:
    """
    Preprocess the data by selecting relevant columns and handling missing values.

    Args:
    data (pd.DataFrame): Loaded data.

    Returns:
    pd.DataFrame: Preprocessed data.

    Raises:
    ValueError: If the data is empty.
    """
    if data.empty:
        raise ValueError("Data is empty")
    try:
        # Select relevant columns
        data = data.dropna()
        logging.info("Data preprocessed")
        return data
    except Exception as e:
        logging.error(f"Error during preprocessing: {e}")
        raise e

def train_model(data: pd.DataFrame) -> LinearRegression:
    """
    Train a linear regression model on the preprocessed data.

    Args:
    data (pd.DataFrame): Preprocessed data.

    Returns:
    LinearRegression: Trained model.

    Raises:
    ValueError: If the data is empty.
    """
    if data.empty:
        raise ValueError("Data is empty")
    try:
        # Split data into training and testing sets
        X_train, X_test, y_train, y_test = train_test_split(data.drop("target", axis=1), data["target"], test_size=0.2, random_state=42)
        # Train linear regression model
        model = LinearRegression()
        model.fit(X_train, y_train)
        logging.info("Model trained")
        return model
    except Exception as e:
        logging.error(f"Error during training: {e}")
        raise e

def evaluate_model(model: LinearRegression, data: pd.DataFrame) -> float:
    """
    Evaluate the performance of the trained model on the preprocessed data.

    Args:
    model (LinearRegression): Trained model.
    data (pd.DataFrame): Preprocessed data.

    Returns:
    float: Mean squared error of the model.

    Raises:
    ValueError: If the data is empty.
    """
    if data.empty:
        raise ValueError("Data is empty")
    try:
        # Make predictions on testing set
        y_pred = model.predict(data.drop("target", axis=1))
        # Calculate mean squared error
        mse = mean_squared_error(data["target"], y_pred)
        logging.info(f"Model evaluated with MSE: {mse}")
        return mse
    except Exception as e:
        logging.error(f"Error during evaluation: {e}")
        raise e

def plot_results(data: pd.DataFrame, model: LinearRegression) -> None:
    """
    Plot the results of the trained model on the preprocessed data.

    Args:
    data (pd.DataFrame): Preprocessed data.
    model (LinearRegression): Trained model.

    Returns:
    None
    """
    try:
        # Make predictions on entire dataset
        y_pred = model.predict(data.drop("target", axis=1))
        # Plot results
        plt.figure(figsize=(10, 6))
        sns.scatterplot(x=data["target"], y=y_pred)
        plt.xlabel("Actual values")
        plt.ylabel("Predicted values")
        plt.title("Model results")
        plt.show()
        logging.info("Results plotted")
    except Exception as e:
        logging.error(f"Error during plotting: {e}")
        raise e
